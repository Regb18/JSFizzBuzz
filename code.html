// entry point
// Get values from user
function getValues() {
    let fizzValue = parseInt(document.getElementById('fizzValue').value)
    let buzzValue = parseInt(document.getElementById('buzzValue').value)
    let stopValue = parseInt(document.getElementById('stopValue').value)

    if (Number.isInteger(fizzValue) && Number.isInteger(buzzValue) && Number.isInteger(stopValue) && fizzValue > 0 && buzzValue > 0 && stopValue >= 0) {

        let numberArray = generateFizzBuzz(stopValue);

        displayFizzBuzz(numberArray, fizzValue, buzzValue);

    } else {
        // if not, tell our user
        Swal.fire(
            {
                icon: 'error',
                title: 'Oops!',
                text: 'Please Enter Numbers Greater than 0'
            }
        )
    }
    // Add error if stop is over 5000
    // Add error if a number is 0
    // Can put function calls at the bottom and use isNaN || to check for numbers

}

// logic function
// Create array of values
function generateFizzBuzz(stop) {

    let numbers = [];

    for(let i = 1; i <= stop; i++) {
        numbers.push(i);
    }

    return numbers;
}


// view function
// Display Values
function displayFizzBuzz(array, fizz, buzz) {

    let tableBody = document.getElementById('results');
    let tableHTML = ""
    
   for (i = 0; i < array.length; i++) {

        let value = array[i];
        let className = ""

        if (value % fizz == 0 && value % buzz == 0) {
            // output FizzBuzz class
            value = 'FizzBuzz';
            className = 'fizzBuzz'

        } else if (value % fizz == 0) {
            // output Fizz class
            value = 'Fizz'
            className = 'fizz'

        } else if (value % buzz == 0) {
            // output Buzz class
            value = 'Buzz'
            className = 'buzz'

        }

        if (i % 5 == 0) {
            tableHTML += "<tr>"
        }
        
        let tableRow = `<td class="${className}">${value}</td>`; 
        tableHTML += tableRow; 
        
        if ((i + 1) % 5 == 0) {
            tableHTML += "</tr>" 
        } 
        

    }

    tableBody.innerHTML = tableHTML;
}

    let tableHTML = ""
    
    for (i = 0; i < array.length; i++) {

        let className =

        i % 5 == 0 ? tableHTML += "<tr>" :

The code is structured in one function

getValues()

This function first takes the values from the html page and converts them all into ints. It then checks to make sure all of the inputs are numbers and that they are greater than zero. If the condition is true we pass the stop value into generateFizzBuzz and get an array and then pass the array, and our fizz and buzz values to output the results. If we don’t pass the condition you’ll get an error and have a chance to try again.

generateFizzBuzz()

This function takes stop as a parameter which we will use to get the correct number of items in the array. First we’ll create an empty array and then use a for loop to iterate through that array 1 index at a time until we reach stop and then exit the array. Inside the loop we’ll add each i value to the array: numbers. Since we always want to start at 1 we can set that as the i value. After that, we’ll return the array numbers back to the calling function.

displayFizzBuzz()

In the Display FizzBuzz function I iterate through the array I’ve created and test each number to see if they are divisible by the fizz and buzz values that the user input. If a number is divisible by both fizz and buzz it is assigned the string value “FizzBuzz” and a classname which is used to change the appearance of the text and table cell it’s in. Otherwise, if the number is equal to fizz or buzz, it will take that string name and class name. The next two of statements are used to ensure the table row is only five entries long before dropping to another row. The code in between there adds our number or FizzBuzz value and it’s class name in HTML format. We then use the tableBody variable which connects to the results ID on our HTML that designates where on the page we want to output the data.
